<script>
    // Firebase Config
    const firebaseConfig = {
        databaseURL: "https://esp32-testing-aec8b-default-rtdb.firebaseio.com" 
    };

    const app = firebase.initializeApp(firebaseConfig);
    const database = firebase.database();

    const deviceId = "ESP32-001";

    function fetchLiveReading() {
        const readingRef = database.ref('readings/' + deviceId);
        readingRef.on('value', (snapshot) => {
            const data = snapshot.val();
            if (data) {
                animateValue('voltage', data.voltage.toFixed(1));
                animateValue('current', data.current.toFixed(4));
                animateValue('power', data.power.toFixed(2));
                animateValue('deviceType', data.deviceType || "Unknown");
                const date = new Date(data.timestamp);
                animateValue('timestamp', date.toLocaleTimeString());
            }
        });
    }

    function animateValue(id, newValue) {
        const el = document.getElementById(id);
        if (el.innerText !== newValue) {
            el.style.transition = 'all 0.5s ease';
            el.style.transform = 'scale(1.2)';
            setTimeout(() => el.style.transform = 'scale(1)', 500);
            el.innerText = newValue;
        }
    }

    fetchLiveReading();
</script>
<script>
    // If you already define these elsewhere, keep only one copy.
    const chatBox = document.getElementById("chatBox");
    const userMessage = document.getElementById("userMessage");
    const sendBtn = document.getElementById("sendBtn");

    // 🔗 Your live-reading endpoint
    const endpoint = '@Url.Action("Latest", "EnergyReadingView")';

    // Small helper used in your UI
    function fmt(val, unit, decimals) {
        if (val === null || val === undefined) return '--';
        return Number(val).toFixed(decimals) + ' ' + unit;
    }

    // ---- Chat ----
    sendBtn.addEventListener("click", sendMessage);
    userMessage.addEventListener("keypress", function (e) {
        if (e.key === "Enter") {
            e.preventDefault();
            sendMessage();
        }
    });

    function appendMessage(sender, message, isError = false) {
        const div = document.createElement("div");
        div.classList.add("chat-message", sender.toLowerCase());
        if (isError) div.classList.add("error");
        div.innerHTML = `<b>${sender}:</b> ${message}`;
        chatBox.appendChild(div);
        chatBox.scrollTop = chatBox.scrollHeight;
    }

    function sendMessage() {
        const message = userMessage.value.trim();
        if (!message) return;

        appendMessage("You", message);
        userMessage.value = "";

        fetch("/Llama/Chat", {
            method: "POST",
            headers: { "Content-Type": "application/json" },
            body: JSON.stringify({ Question: message }) // must match C# model property
        })
        .then(res => {
            if (!res.ok) throw new Error("Network response was not ok");
            return res.json();
        })
        .then(data => {
            if (data.error) {
                appendMessage("Error", data.error, true);
            } else {
                // data.reply expected from your Flask API
                appendMessage("AI", data.reply ?? JSON.stringify(data));
            }
        })
        .catch(err => {
            appendMessage("Error", err.message, true);
        });
    } // ✅ CLOSE sendMessage properly

    // ---- Live readings + analysis ----
    async function refresh() {
        try {
            const res = await fetch(endpoint, { cache: 'no-store' });
            if (!res.ok) return;

            const d = await res.json();
            if (!d.hasData) {
                document.getElementById('lastTime').innerText = 'No data yet';
                return;
            }

            document.getElementById('voltageVal').innerText = fmt(d.voltage, 'V', 1);
            document.getElementById('currentVal').innerText = fmt(d.current, 'A', 1);
            document.getElementById('powerVal').innerText   = fmt(d.power, 'kW', 2);
            document.getElementById('lastTime').innerText   = new Date(d.timestamp).toLocaleString();

            // Show analyzing while we fetch
            const levelEl = document.getElementById('consumptionLevel');
            const tipEl = document.getElementById('consumptionTip');
            if (levelEl && tipEl) {
                levelEl.innerText = 'Analyzing...';
                tipEl.innerText = 'Analyzing...';
            }

            // 🔑 Send reading to backend for analysis
            const analysisRes = await fetch("/Llama/AnalyzeReading", {
                method: "POST",
                headers: { "Content-Type": "application/json" },
                body: JSON.stringify({
                    voltage: d.voltage,
                    current: d.current,
                    power: d.power
                })
            });

            // If the controller returns plain text instead of JSON, this will throw
            let analysis;
            try {
                analysis = await analysisRes.json();
            } catch {
                // Fallback: show raw text
                const raw = await analysisRes.text();
                if (levelEl) levelEl.innerText = '—';
                if (tipEl) tipEl.innerText = raw || 'No tip available';
                return;
            }

            // Supports both shapes:
            // 1) { consumptionLevel, tip }
            // 2) { reply } from AI
            const level = analysis.consumptionLevel ?? '—';
            const tip = analysis.tip ?? analysis.reply ?? 'No tip available';

            if (levelEl) levelEl.innerText = level;
            if (tipEl) tipEl.innerText = tip;

        } catch (e) {
            console.warn(e);
            const levelEl = document.getElementById('consumptionLevel');
            const tipEl = document.getElementById('consumptionTip');
            if (levelEl) levelEl.innerText = 'Unavailable';
            if (tipEl) tipEl.innerText = 'Could not fetch analysis';
        }
    }

    // 🔁 Actually run it
    refresh();
    //setInterval(refresh, 3000);
</script>
